<h4> how to use </h4>
<pre>
  <code>
      <script type="text/javascript" src="pieChart.js"></script>
      <script type="text/javascript">
        pieChart({
          fillPercent: 50,
          radius: 90,
          stroke: 40,
          backgroundStrokeColor: "teal",
          foregroundStrokeColor: "orange",
          container: document.getElementById('im_a_div')
        });
      </script>
  </code>
